  // 1900-01-01 JD=2415021 → 甲戌=index(0,10)=10，但DZ週期12，TG週期10

  // 反查：哪個五行「生」dayWX（印星）
  const shengReverse = {};
  for(const [k,v] of Object.entries(sheng)) shengReverse[v]=k;
  // 反查：哪個五行「剋」dayWX（官殺）
  const keReverse = {};
  for(const [k,v] of Object.entries(ke)) keReverse[v]=k;

  let yong=[], ji=[];
  if(strengthLabel==='身強'){
    // 洩耗為用：食傷（我生）、財（我剋）、官殺（剋我）
    // 忌：比劫（同氣）、印星（生我）
    yong = [sheng[dayWX], ke[dayWX], keReverse[dayWX]].filter((v,i,a)=>a.indexOf(v)===i);
    ji   = [dayWX, shengReverse[dayWX]];
  } else if(strengthLabel==='身弱'){
    // 幫扶為用：比劫（同氣）、印星（生我）
    // 忌：食傷（我生洩身）、財（我剋耗身）
    yong = [dayWX, shengReverse[dayWX]];
    ji   = [sheng[dayWX], ke[dayWX]];
  } else {
    yong = [dayWX];
    ji   = [];
  }
  // 去重
  yong = [...new Set(yong)];
  ji   = [...new Set(ji.filter(x=>!yong.includes(x)))];
  return {yong, ji};
}

const SHENG = {木:'火',火:'土',土:'金',金:'水',水:'木'};

// 判斷某五行對日主的關係（+2強助 +1輕助 0中性 -1輕剋 -2強剋）
function wxRelation(wx, dayWX, yongshen){
  const isYong = yongshen.yong.includes(wx);
  const isJi   = yongshen.ji.includes(wx);
  if(isYong) return 2;
  if(isJi)   return -2;
  // 次要判斷
  if(wx === dayWX) return 1;           // 同氣，算輕助
  if(SHENG[wx] === dayWX) return 1;   // 生我，算輕助
  if(KE[wx]   === dayWX) return -1;   // 剋我，算輕剋
  if(SHENG[dayWX] === wx) return -1;  // 我生（洩），算輕剋
  return 0;
}

// 計算一柱（tg+dz）對命局的整體分數 [-4 ~ 4]
function calcGZScore(tg, dz, dayTG, yongshen){
  const dayWX = TG_WX[dayTG];
  const tgWX  = TG_WX[tg];
  const dzWX  = DZ_WX[dz];
  const hiddenWX = (DZ_HIDDEN[dz]||[]).map(h=>TG_WX[h]);

  let score = 0;
  score += wxRelation(tgWX, dayWX, yongshen) * 1.2;  // 天干權重稍高
  score += wxRelation(dzWX, dayWX, yongshen) * 1.0;
  // 藏干主氣
  if(hiddenWX.length) score += wxRelation(hiddenWX[0], dayWX, yongshen) * 0.6;
  return score;
}

function calcDomainScore(tg, dz, dayTG, yongshen){
  const dayWX = TG_WX[dayTG];
  const tgWX  = TG_WX[tg];
  const dzWX  = DZ_WX[dz];
  const allWX = [tgWX, dzWX];

  const scores = {};
  for(const [domain, weights] of Object.entries(DOMAIN_WX_WEIGHT)){
    let s = 0;
    for(const wx of allWX){
      const rel = wxRelation(wx, dayWX, yongshen);  // -2~2
      s += rel * (weights[wx]||0.5);
    }
    // 歸一到 0-100
    scores[domain] = Math.round(Math.max(0, Math.min(100, (s/4 + 0.5) * 100)));
  }
  return scores;
}

  const dayTG    = _currentDayTG;
  const yongshen = _currentYongshen;

  const dyTGWX = TG_WX[dayun.tg];
  const dyDZWX = DZ_WX[dayun.dz];
  const lyTGWX = TG_WX[lyTG];

  else if(pct>=44){ scoreLevel='平穩'; scoreColor='var(--muted)'; }
  else if(pct>=28){ scoreLevel='小凶'; scoreColor='var(--purple)'; }
  else             { scoreLevel='大凶'; scoreColor='var(--accent)'; }

  // ── Badge：顯示大運干支 + 流年干支 ──
  const WX_COLOR = {木:'var(--木)',火:'var(--火)',土:'var(--土)',金:'var(--金)',水:'var(--水)'};

  const lyPct = Math.round(Math.max(0,Math.min(100,(lyScore/4+0.5)*100)));

    `${dayun.tg}${dayun.dz} 大運（${dayun.startAge}歲）× ${lyYear} ${lyTG}${lyDZ} 流年`;

    `大運 <b>${dayun.tg}${dayun.dz}</b>（${dyTGWX}/${dyDZWX}）${relLabel[dyRel]} ${dyPct}分 &nbsp;＋&nbsp; ` +

  fill.style.background = scoreColor;

  const dyDomain = calcDomainScore(dayun.tg, dayun.dz, dayTG, yongshen);
  const lyDomain = calcDomainScore(lyTG, lyDZ, dayTG, yongshen);
  const domainsEl = document.getElementById('advice-domains');

      <div class="domain-chip ${cls}">
        <span class="d-icon">${meta.icon}</span>
        <span class="d-label">${meta.label}</span>
        <span class="d-val">${arrow} ${s}</span>
      </div>`;
  }

  // ── 宜忌建議：以「流年為主、大運為底」合併去重 ──
  // 流年關係決定主要宜忌，大運關係補充
  const lyTGRel = rel(lyTGWX);
  const lyDZRel = rel(lyDZWX);
  const dyTGRel = rel(dyTGWX);
  const dyDZRel = rel(dyDZWX);

    ...getAdviceList(lyTGWX,'do', lyTGRel),
    ...getAdviceList(lyDZWX,'do', lyDZRel),
    ...getAdviceList(dyTGWX,'do', dyTGRel),

    ...getAdviceList(lyTGWX,'dont', lyTGRel),
    ...getAdviceList(lyDZWX,'dont', lyDZRel),
    ...getAdviceList(dyTGWX,'dont', dyTGRel),

  const dontFinal = uniq(dontList);

      </ul>

  // ── 五行對照：同時呈現大運與流年 ──
  const awaRows = document.getElementById('awa-rows');
  awaRows.innerHTML = '';
  const wxList = ['木','火','土','金','水'];
  const dyWXSet = [dyTGWX, dyDZWX, ...(DZ_HIDDEN[dayun.dz]||[]).map(h=>TG_WX[h])];
  const lyWXSet = [lyTGWX, lyDZWX, ...(DZ_HIDDEN[lyDZ]||[]).map(h=>TG_WX[h])];

  for(const wx of wxList){
    const inDY  = dyWXSet.includes(wx);
    const inLY  = lyWXSet.includes(wx);
    const isYong= yongshen.yong.includes(wx);
    const isJi  = yongshen.ji.includes(wx);
    // 強度：流年出現比大運出現更重要
    const strength = (inLY?2:0) + (inDY?1:0);
    const barPct = [0,25,60,90][strength] * (isYong?1:isJi?0.95:0.55);
    const barColor = WX_COLOR[wx];
    const srcLabel = inLY && inDY ? '大運＋流年' : inLY ? '流年' : inDY ? '大運' : '未出現';
    const tagText  = !inLY && !inDY ? '未出現'
                   : isYong ? `用神得令 ✦ (${srcLabel})`
                   : isJi   ? `忌神當令 ✗ (${srcLabel})`
                   :          `中性 (${srcLabel})`;
    const tagBg    = !inLY&&!inDY?'var(--border)':isYong?'var(--木-bg)':isJi?'var(--火-bg)':'var(--bg)';
    const tagColor = !inLY&&!inDY?'var(--muted)' :isYong?'var(--teal)' :isJi?'var(--accent)':'var(--ink)';
    awaRows.innerHTML += `
      <div class="awa-row">
        <div class="awa-wx" style="color:${barColor}">${wx}</div>
        <div class="awa-bar-track">
          <div class="awa-bar-fill" style="width:${Math.round(barPct)}%;background:${barColor};opacity:${(inLY||inDY)?1:.25}"></div>
        </div>
        <div class="awa-tag" style="background:${tagBg};color:${tagColor};border:1px solid ${inLY||inDY?(isYong?'#a8d8c0':isJi?'#e8b8b8':'var(--border)'):'var(--border)'}">${tagText}</div>
      </div>`;
  }

  card.classList.add('visible');
  setTimeout(()=>card.scrollIntoView({behavior:'smooth', block:'nearest'}), 100);
}

// ── 取得今日/本月干支的個人化建議 ──
function getLifeAdvice(tg, dz, dayTG, yongshen, strength){
  const dayWX = TG_WX[dayTG];
  const tgWX  = TG_WX[tg];
  const dzWX  = DZ_WX[dz];
  // 主五行取天干，次五行取地支
  const mainWX  = tgWX;
  const subWX   = dzWX;
  const rel     = wxRelation(mainWX, dayWX, yongshen);
  const relKey  = rel>=1?'yong':rel<=-1?'ji':'neutral';
  const strKey  = strength; // '身強'|'身弱'|'中和'
  const normStr = strKey==='中和'?'身弱':strKey; // 中和時用身弱邏輯作保守建議

  const result = {};
  for(const cat of ['食','衣','住','行']){
    const db = LIFE_DB[cat][mainWX];
    let tips;
    if(relKey==='neutral'){
      tips = db.neutral || [];
    } else {
      tips = (db[relKey] && db[relKey][normStr]) || db.neutral || [];
    }
    // 補充地支五行建議（若地支與天干不同五行）
    let subTips = [];
    if(subWX !== mainWX){
      const subRel = wxRelation(subWX, dayWX, yongshen);
      const subRelKey = subRel>=1?'yong':subRel<=-1?'ji':'neutral';
      const subDb = LIFE_DB[cat][subWX];
      if(subRelKey==='neutral'){
        subTips = (subDb.neutral || []).slice(0,1);
      } else {
        subTips = ((subDb[subRelKey] && subDb[subRelKey][normStr]) || subDb.neutral || []).slice(0,1);
      }
    }
    // 合併，天干建議前2條 + 地支建議1條，共最多3條
    result[cat] = [...tips.slice(0,2), ...subTips].slice(0,3);
  }
  return result;
}

// ── 計算分數並轉換文字 ──
function scoreToText(pct){
  if(pct>=80) return {lv:'大吉',color:'var(--gold)'};
  if(pct>=65) return {lv:'小吉',color:'var(--teal)'};
  if(pct>=48) return {lv:'平穩',color:'var(--muted)'};
  if(pct>=32) return {lv:'留意',color:'var(--purple)'};
  return {lv:'謹慎',color:'var(--accent)'};
}

  const dayTG    = _currentDayTG;

  const dayWX    = TG_WX[dayTG];
  const isDay    = (dTG !== null);

  // ══════════════════════════════════════════════════
  // 核心：把流月五行 + 流日五行合成「當期五行環境向量」
  // 向量 = 每個五行在當期環境中的強度（0~1）
  // 月份貢獻 40%，日期貢獻 60%
  // 每柱有天干五行（權重1.2）+ 地支五行（權重1.0）+ 地支藏干主氣（權重0.6）
  // ══════════════════════════════════════════════════
  function buildWXVector(tg, dz, weight){
    const vec = {木:0,火:0,土:0,金:0,水:0};
    vec[TG_WX[tg]]  += 1.2 * weight;
    vec[DZ_WX[dz]]  += 1.0 * weight;
    const hidden = DZ_HIDDEN[dz] || [];
    if(hidden.length) vec[TG_WX[hidden[0]]] += 0.6 * weight;
    return vec;
  }
  function addVec(a, b){ const r={};for(const k of Object.keys(a)) r[k]=(a[k]||0)+(b[k]||0);return r; }

  let envVec;
  if(!isDay){
    // 月模式：只有流月
    envVec = buildWXVector(mTG, mDZ, 1.0);
  } else {
    // 日模式：流月 40% + 流日 60%
    const mv = buildWXVector(mTG, mDZ, 0.4);
    const dv = buildWXVector(dTG, dDZ, 0.6);
    envVec = addVec(mv, dv);
  }

  // 把向量正規化（最大值設為1），方便後續判斷強度
  const maxV = Math.max(...Object.values(envVec), 0.001);
  const normVec = {};
  for(const k of Object.keys(envVec)) normVec[k] = envVec[k] / maxV;

  // ══════════════════════════════════════════════════
  // 綜合評分：每個五行的環境強度 × 對日主的用神關係
  // ══════════════════════════════════════════════════
  let rawScore = 0, rawMax = 0;
  for(const [wx, intensity] of Object.entries(envVec)){
    const rel = wxRelation(wx, dayWX, yongshen); // -2~+2
    rawScore += rel * intensity;
    rawMax   += 2  * intensity;
  }
  const pct = Math.round(Math.max(0, Math.min(100, (rawScore/rawMax + 1) / 2 * 100)));
  const {lv, color} = scoreToText(pct);

  // ══════════════════════════════════════════════════
  // 找出「主導五行」（環境向量最強的）和「次要五行」
  // ══════════════════════════════════════════════════
  const sorted = Object.entries(envVec).sort((a,b)=>b[1]-a[1]);
  const mainWX = sorted[0][0];
  const subWX  = sorted[1][0];
  const mainRel = wxRelation(mainWX, dayWX, yongshen);
  const subRel  = wxRelation(subWX,  dayWX, yongshen);
  const mainRelKey = mainRel>=1?'yong':mainRel<=-1?'ji':'neutral';

    document.getElementById('detail-gz-dz').innerHTML =
      `<span style="font-size:20px;font-weight:900;color:${WX_COLOR_MAP[dTGWX]}">${dTG}${dDZ}</span>`;
  } else {
    const mTGWX=TG_WX[mTG];
    document.getElementById('detail-gz-tg').textContent = mTG;
    document.getElementById('detail-gz-tg').style.color = WX_COLOR_MAP[mTGWX];
    document.getElementById('detail-gz-dz').textContent = mDZ;
    document.getElementById('detail-gz-dz').style.color = WX_COLOR_MAP[DZ_WX[mDZ]];
  }

  document.getElementById('detail-title').textContent = titleText;

  // ── 說明列：五行環境組成 ──
  const relCN = rel => rel>=1?'用神▲':rel<=-1?'忌神▼':'中性';
  const wxEnvDesc = sorted.slice(0,3)

  // 食衣住行：根據主導五行 × 關係 × 強弱給建議

  const doList   = [...getAdviceList(mainWX,'do',mainRelKey), ...getAdviceList(subWX,'do',subRelKey)];

  _dailyMode = mode;
  document.querySelectorAll('.daily-nav-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
  document.getElementById('daily-calendar').style.display = mode==='day'?'block':'none';
  if(mode==='day'){

// ── 日曆 ──
function renderCalendar(year, month){
  const grid = document.getElementById('day-grid');
  grid.innerHTML = '';
  const today = new Date();
  const firstDay = new Date(year, month-1, 1).getDay();
  const daysInMonth = new Date(year, month, 0).getDate();

  // 空格
  for(let i=0;i<firstDay;i++){
    const e=document.createElement('div');
    e.className='day-cell empty';
    grid.appendChild(e);
  }

  for(let d=1; d<=daysInMonth; d++){
    const gz = dayGZ(year, month, d);
    const tgWX = TG_WX[gz.tg];
    const rel = _currentYongshen ? wxRelation(tgWX, TG_WX[_currentDayTG], _currentYongshen) : 0;
    const relCls = rel>=1?'day-yong':rel<=-1?'day-ji':'day-neutral';
    const isToday = (d===today.getDate()&&month===today.getMonth()+1&&year===today.getFullYear());
    const cell = document.createElement('div');
    cell.className = `day-cell ${relCls}${isToday?' today':''}`;
    cell.innerHTML = `
      <div class="dc-date">${d}</div>
      <div class="dc-gz">${gz.tg}${gz.dz}</div>
      <div class="dc-dot"></div>`;
    cell.addEventListener('click', ()=>{
      document.querySelectorAll('.day-cell').forEach(c=>c.classList.remove('selected'));
      cell.classList.add('selected');
      const mgz = monthGZ(year, month);
      const weekNames=['日','一','二','三','四','五','六'];
      const wk = new Date(year,month-1,d).getDay();
      renderDetailCard(mgz.tg, mgz.dz, gz.tg, gz.dz,
        `${year}年${month}月${d}日（週${weekNames[wk]}）${gz.tg}${gz.dz}日`,
        '');
    });
    grid.appendChild(cell);
  }
}
